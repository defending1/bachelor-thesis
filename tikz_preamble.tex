% Try to use the same font setup as Thesis.tex. The snippet below mirrors the
% conditional used in Thesis.tex/mimosis.cls: when compiling with XeTeX or LuaTeX
% it uses fontspec; otherwise it falls back to the PDFLaTeX packages.
\RequirePackage{iftex}
\newif\ifxetexorluatex\xetexorluatexfalse
\ifxetex
  \xetexorluatextrue
\fi
\ifluatex
  \xetexorluatextrue
\fi

\ifxetexorluatex
  % For XeLaTeX/LuaLaTeX: attempt to set the same fonts as the thesis
  \usepackage{fontspec}
  \usepackage{unicode-math}
  % These \IfFontExistsTF checks are not available without expl3; keep simple
  % and rely on system availability when using Xe/Lua engines.
  \IfFontExistsTF{EB Garamond}{%
    \setmainfont{EB Garamond}%
  }{%
    \setmainfont{DejaVu Serif}%
  }
  \IfFontExistsTF{Garamond Math}{%
    \setmathfont{Garamond Math}%
  }{%
    \setmathfont{DejaVu Math TeX Gyre}%
  }
  \IfFontExistsTF{STIX Two Math}{%
    \setmathfont{STIX Two Math}[range={\sharp,\natural,\flat,\clubsuit,\spadesuit,\checkmark}]%
  }{}
  \IfFontExistsTF{Source Code Pro}{%
    \setmonofont[Scale=MatchLowercase]{Source Code Pro}%
  }{%
    \setmonofont[Scale=MatchLowercase]{DejaVu Sans Mono}%
  }
\else
  % For pdfLaTeX: use the same packages as the thesis
  \usepackage{setspace}
  \usepackage[lf]{ebgaramond}
  \usepackage[oldstyle,scale=0.7]{sourcecodepro}
  \singlespacing
\fi
